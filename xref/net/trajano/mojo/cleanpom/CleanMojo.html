27  /**
28   * Cleans the POM.
29   */
30  @Mojo(name = "clean", defaultPhase = LifecyclePhase.VERIFY, threadSafe = true)
31  public class CleanMojo extends AbstractMojo {
32      /**
33       * Resource bundle.
34       */
35      private static final ResourceBundle R = ResourceBundle
36              .getBundle("META-INF/Messages");
37  
38      /**
39       * Build context.
40       */
41      @Component
42      private BuildContext buildContext;
43  
44      /**
45       * The POM file to update.
46       */
47      @Parameter(defaultValue = "${basedir}/pom.xml", required = false)
48      private File pomFile;
49  
50      /**
51       * Transformer.
52       */
53      private final Transformer transformer;
54  
55      /**
56       * Initializes the transfomer.
57       */
58      public CleanMojo() {
59          try {
60              final InputStream xsltStream = getClass().getResourceAsStream(
61                      "/META-INF/pom-clean.xslt");
62              transformer = TransformerFactory.newInstance().newTransformer(
63                      new StreamSource(xsltStream));
64              xsltStream.close();
65          } catch (final TransformerConfigurationException e) {
66              throw new IllegalStateException(e);
67          } catch (final IOException e) {
68              throw new IllegalStateException(e);
69          }
70      }
71  
72      /**
73       * Builds the temporary file based on the {@link #pomFile}.
74       *
75       * @return temporary file
76       * @throws MojoExecutionException
77       */
78      private File buildTempFile() throws MojoExecutionException {
79          final File tempFile;
80          try {
81              tempFile = File.createTempFile("pom", "xml");
82              tempFile.delete();
83          } catch (final IOException e) {
84              throw new MojoExecutionException(e.getMessage(), e);
85          }
86          try {
87              if (pomFile == null) {
88                  pomFile = new File("pom.xml");
89              }
90              FileUtils.copyFile(pomFile, tempFile);
91          } catch (final IOException e) {
92              throw new MojoExecutionException(format(R.getString("copyfail"),
93                      pomFile, tempFile), e);
94          }
95          return tempFile;
96      }
97  
98      /**
99       * Performs the cleanup.
100      *
101      * @throws MojoExecutionException
102      *             problem with the execution.
103      */
104     @Override
105     public void execute() throws MojoExecutionException {
106         final File tempFile = buildTempFile();
107         transform(tempFile, pomFile);
108     }
109 
110     /**
111      * Performs the transformation.
112      *
113      * @param sourceFile
114      *            source file
115      * @param targetFile
116      *            target file
117      * @throws MojoExecutionException
118      *             problem with the execution.
119      */
120     private void transform(final File sourceFile, final File targetFile)
121             throws MojoExecutionException {
122         try {
123             final OutputStream outputStream = buildContext
124                     .newFileOutputStream(targetFile);
125             transformer.transform(new StreamSource(sourceFile),
126                     new StreamResult(outputStream));
127             getLog().debug(format(R.getString("donecleaning"), targetFile));
128             sourceFile.delete();
129             outputStream.close();
130         } catch (final TransformerException e) {
131             throw new MojoExecutionException(format(
132                     R.getString("transformfail"), sourceFile), e);
133         } catch (final IOException e) {
134             throw new MojoExecutionException(format(
135                     R.getString("transformfail"), sourceFile), e);
136         }
137     }
