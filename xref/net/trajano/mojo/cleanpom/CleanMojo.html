22  /**
23   * Cleans the POM.
24   */
25  @Mojo(name = "clean", defaultPhase = LifecyclePhase.VERIFY, threadSafe = true)
26  public class CleanMojo extends AbstractMojo {
27      /**
28       * Resource bundle.
29       */
30      private static final ResourceBundle R = ResourceBundle
31              .getBundle("META-INF/Messages");
32  
33      /**
34       * The POM file to update.
35       */
36      @Parameter(defaultValue = "${basedir}/pom.xml", required = false)
37      private File pomFile;
38  
39      /**
40       * Builds the temporary file based on the {@link #pomFile}.
41       *
42       * @return temporary file
43       * @throws MojoExecutionException
44       */
45      private File buildTempFile() throws MojoExecutionException {
46          final File tempFile;
47          try {
48              tempFile = File.createTempFile("pom", "xml");
49              tempFile.delete();
50          } catch (final IOException e) {
51              throw new MojoExecutionException(e.getMessage(), e);
52          }
53          try {
54              if (pomFile == null) {
55                  pomFile = new File("pom.xml");
56              }
57              FileUtils.copyFile(pomFile, tempFile);
58          } catch (final IOException e) {
59              throw new MojoExecutionException(format(R.getString("copyfail"),
60                      pomFile, tempFile), e);
61          }
62          return tempFile;
63      }
64  
65      /**
66       * Performs the cleanup.
67       */
68      public void execute() throws MojoExecutionException {
69          final File tempFile = buildTempFile();
70          try {
71              final Transformer t = TransformerFactory.newInstance()
72                      .newTransformer(
73                              new StreamSource(getClass().getResourceAsStream(
74                                      "/META-INF/pom-clean.xslt")));
75              t.transform(new StreamSource(tempFile), new StreamResult(pomFile));
76              getLog().debug(format(R.getString("donecleaning"), pomFile));
77              tempFile.delete();
78          } catch (final TransformerException e) {
79              throw new MojoExecutionException(format(
80                      R.getString("transformfail"), tempFile), e);
81          }
82      }
