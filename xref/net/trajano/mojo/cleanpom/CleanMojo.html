32  /**
33   * Cleans the POM.
34   */
35  @Mojo(name = "clean",
36      defaultPhase = LifecyclePhase.PREPARE_PACKAGE,
37      threadSafe = true)
38  public class CleanMojo extends AbstractMojo {
39  
40      /**
41       * Resource bundle.
42       */
43      private static final ResourceBundle R = ResourceBundle.getBundle("META-INF/Messages");
44  
45      /**
46       * Build context.
47       */
48      @Component
49      private BuildContext buildContext;
50  
51      /**
52       * The POM file to update.
53       */
54      @Parameter(defaultValue = "${basedir}/pom.xml",
55          property = "cleanpom.pomFile",
56          required = false)
57      private File pomFile;
58  
59      /**
60       * List of XSLT files as a comma separated value. Only one of xsltFiles or
61       * xsltFileList is allowed.
62       */
63      @Parameter(defaultValue = "/META-INF/pom-clean.xslt",
64          required = false,
65          property = "cleanpom.xsltFileList")
66      private String xsltFileList;
67  
68      /**
69       * List of XSLT files.
70       */
71      @Parameter(required = false)
72      private String[] xsltFiles;
73  
74      /**
75       * Builds the handler chain.
76       *
77       * @param tf
78       *            SAX transformer factory
79       * @param outputStream
80       *            output stream
81       * @return handler chain.
82       * @throws IOException
83       *             I/O error when reading the XSLT file.
84       * @throws TransformerException
85       *             problem building the templates.
86       */
87      private TransformerHandler buildHandlerChain(
88          final SAXTransformerFactory tf,
89          final OutputStream outputStream)
90          throws IOException,
91          TransformerException {
92  
93          TransformerHandler handler = null;
94          TransformerHandler lastHandler = null;
95  
96          for (final String xsltFile : xsltFiles) {
97              final InputStream xsltStream = getClass().getResourceAsStream(
98                  xsltFile.charAt(0) == '/' ? xsltFile : "/META-INF/" + xsltFile);
99              // The stream source needs to be defined here.
100             final TransformerHandler currentHandler = tf.newTransformerHandler(new StreamSource(xsltStream)); // NOPMD
101             xsltStream.close();
102             if (lastHandler != null) {
103                 // The result object needs to be created here.
104                 lastHandler.setResult(new SAXResult(currentHandler)); // NOPMD
105                 lastHandler = currentHandler;
106             } else {
107                 lastHandler = currentHandler;
108                 handler = currentHandler;
109             }
110         }
111         lastHandler.setResult(new StreamResult(outputStream));
112         return handler;
113     }
114 
115     /**
116      * Builds the temporary file based on the {@link #pomFile}.
117      *
118      * @return temporary file
119      * @throws MojoExecutionException
120      *             wraps {@link IOException}
121      */
122     private File buildTempFile()
123         throws MojoExecutionException {
124 
125         final File tempFile;
126         try {
127             tempFile = File.createTempFile("pom", "xml");
128             tempFile.delete();
129         } catch (final IOException e) {
130             throw new MojoExecutionException(e.getMessage(), e);
131         }
132         try {
133             if (pomFile == null) {
134                 pomFile = new File("pom.xml");
135             }
136             FileUtils.copyFile(pomFile, tempFile);
137         } catch (final IOException e) {
138             throw new MojoExecutionException(format(R.getString("copyfail"), pomFile, tempFile), e);
139         }
140         return tempFile;
141     }
142 
143     /**
144      * Performs the cleanup.
145      *
146      * @throws MojoExecutionException
147      *             problem with the execution.
148      */
149     @Override
150     public void execute()
151         throws MojoExecutionException {
152 
153         final File tempFile = buildTempFile();
154         transform(tempFile, pomFile);
155     }
156 
157     /**
158      * Performs the transformation.
159      *
160      * @param sourceFile
161      *            source file
162      * @param targetFile
163      *            target file
164      * @throws MojoExecutionException
165      *             problem with the execution.
166      */
167     private void transform(final File sourceFile,
168         final File targetFile)
169         throws MojoExecutionException {
170 
171         try {
172             final SAXTransformerFactory tf = (SAXTransformerFactory) TransformerFactory.newInstance();
173 
174             if (xsltFiles == null) {
175                 if (xsltFileList != null) {
176                     xsltFiles = StringUtils.split(xsltFileList, ",");
177                 } else {
178                     xsltFiles = new String[] {
179                         "/META-INF/pom-clean.xslt"
180                     };
181                 }
182             }
183             final OutputStream outputStream = buildContext.newFileOutputStream(targetFile);
184             final TransformerHandler handler = buildHandlerChain(tf, new EolNormalizingStream(outputStream));
185             final Transformer transformer = tf.newTransformer();
186             transformer.transform(new StreamSource(sourceFile), new SAXResult(handler));
187             getLog().debug(format(R.getString("donecleaning"), targetFile));
188             sourceFile.delete();
189             outputStream.close();
190         } catch (final TransformerException e) {
191             throw new MojoExecutionException(format(
192                 R.getString("transformfail"), sourceFile), e);
193         } catch (final IOException e) {
194             throw new MojoExecutionException(format(
195                 R.getString("transformfailio"), sourceFile), e);
196         }
197     }
