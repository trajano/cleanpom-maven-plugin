39  /**
40   * Cleans XML sources in general. Tied to the {@link LifecyclePhase#INITIALIZE}
41   * so sources are cleaned up before doing anything else.
42   */
43  @Mojo(name = "clean-xml",
44      defaultPhase = LifecyclePhase.INITIALIZE,
45      threadSafe = true)
46  public class CleanXmlMojo extends AbstractMojo {
47  
48      /**
49       * Resource bundle.
50       */
51      private static final ResourceBundle R = ResourceBundle.getBundle("META-INF/Messages");
52  
53      /**
54       * Build context.
55       */
56      @Component
57      private BuildContext buildContext;
58  
59      /**
60       * List of XML filesets to process. This defaults to
61       * <code>src/main/*.[xml|xsd|xslt]</code> and <code>src/site/*.xml</code> .
62       */
63      @Parameter(required = false)
64      private FileSet[] xmlFileSets;
65  
66      /**
67       * Performs the cleanup.
68       *
69       * @throws MojoExecutionException
70       *             problem with the execution.
71       */
72      @Override
73      public void execute() throws MojoExecutionException {
74  
75          if (xmlFileSets == null) {
76              xmlFileSets = new FileSet[2];
77              xmlFileSets[0] = new FileSet();
78              xmlFileSets[0].setDirectory("src/main");
79              xmlFileSets[0].addInclude("**/*.xml");
80              xmlFileSets[0].addInclude("**/*.xsd");
81              xmlFileSets[0].addInclude("**/*.xslt");
82              xmlFileSets[1] = new FileSet();
83              xmlFileSets[1].setDirectory("src/site");
84              xmlFileSets[1].addInclude("**/*.xml");
85          }
86  
87          final File tempFile;
88          try {
89              tempFile = File.createTempFile("temp", "xml");
90          } catch (final IOException e) {
91              throw new MojoExecutionException(e.getMessage(), e);
92          }
93          for (final FileSet xmlFiles : xmlFileSets) {
94              final File dir = new File(xmlFiles.getDirectory());
95              if (!dir.exists()) {
96                  continue;
97              }
98              final Scanner scanner = buildContext.newScanner(dir, false);
99              scanner.setIncludes(xmlFiles.getIncludes().toArray(new String[0]));
100             scanner.scan();
101             for (final String includedFile : scanner.getIncludedFiles()) {
102                 final File file = new File(scanner.getBasedir(), includedFile);
103 
104                 try {
105                     FileUtils.copyFile(file, tempFile);
106                     transform(tempFile, file);
107                 } catch (final IOException e) {
108                     throw new MojoExecutionException(format(R.getString("copyfail"), file, tempFile), e);
109                 }
110 
111             }
112         }
113         tempFile.delete();
114     }
115 
116     /**
117      * Performs the transformation.
118      *
119      * @param sourceFile
120      *            source file
121      * @param targetFile
122      *            target file
123      * @throws MojoExecutionException
124      *             problem with the execution.
125      */
126     private void transform(final File sourceFile,
127         final File targetFile) throws MojoExecutionException {
128 
129         try {
130             final SAXParserFactory spf = SAXParserFactory.newInstance();
131             spf.setNamespaceAware(true);
132             final XMLReader xmlReader = spf.newSAXParser().getXMLReader();
133             final DtdResolver resolver = new DtdResolver();
134             xmlReader.setEntityResolver(resolver);
135             xmlReader.setContentHandler(resolver);
136 
137             final FileInputStream source = new FileInputStream(sourceFile);
138             xmlReader.parse(new InputSource(source));
139             source.close();
140 
141             final SAXTransformerFactory tf = (SAXTransformerFactory) TransformerFactory.newInstance();
142             final OutputStream outputStream = buildContext.newFileOutputStream(targetFile);
143             final Transformer transformer;
144             if (resolver.isDtdPresent()) {
145                 transformer = tf.newTransformer(new StreamSource(Thread.currentThread().getContextClassLoader().getResourceAsStream("META-INF/clean-with-dtd.xslt")));
146                 transformer.setOutputProperty("{http://xml.apache.org/xalan}line-separator", "\n");
147                 transformer.setOutputProperty(OutputKeys.DOCTYPE_PUBLIC, resolver.getPublicId());
148                 transformer.setOutputProperty(OutputKeys.DOCTYPE_SYSTEM, resolver.getSystemId());
149             } else {
150                 transformer = tf.newTransformer(new StreamSource(Thread.currentThread().getContextClassLoader().getResourceAsStream("META-INF/clean.xslt")));
151             }
152             transformer.transform(new StreamSource(sourceFile), new StreamResult(new EolNormalizingStream(outputStream)));
153             getLog().debug(format(R.getString("donecleaning"), targetFile));
154             sourceFile.delete();
155             outputStream.close();
156         } catch (final SAXException e) {
157             throw new MojoExecutionException(format(Messages.TRANSFORM_FAIL, targetFile), e);
158         } catch (final TransformerException e) {
159             throw new MojoExecutionException(format(Messages.TRANSFORM_FAIL, targetFile), e);
160         } catch (final IOException e) {
161             throw new MojoExecutionException(format(Messages.TRANSFORM_FAIL_IO, targetFile), e);
162         } catch (final ParserConfigurationException e) {
163             throw new MojoExecutionException(format(Messages.TRANSFORM_FAIL, targetFile), e);
164         }
165     }
